function countOnes(n) {
    let count = 0;
    while (n > 0) {

            count++;
        }
        n = Math.floor(n / 2);
    }
    return count;

    def countBits(self, n: int) -> List[int]:
        ans = [0] * (n + 1)
        for i in range(1, n + 1):
            ans[i] = self.countOnes(i)
